    <script>

        // Helper to get number value from input

        const getNum = id => parseFloat(document.getElementById(id).value) || 0;


        // Formatting helper

        const formatCurrency = (num) => new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', minimumFractionDigits: 0, maximumFractionDigits: 0 }).format(num);


        function calculate() {

            // --- INPUTS ---

            const arv = getNum('arv');

            const currentValue = getNum('currentValue');

            const repairs = getNum('repairs');

            const closingCosts = getNum('closingCosts');


            const rent = getNum('rent');

            const taxes = getNum('taxes');

            const insurance = getNum('insurance');

            const vacancyPercent = getNum('vacancy') / 100;

            const maintenancePercent = getNum('maintenance') / 100;

            const propManagementPercent = getNum('propManagement') / 100;


            const desiredCashflow = getNum('desiredCashflow');

            const sellerFinanceRate = getNum('sellerFinanceRate');

            const sellerFinanceDownPayment = getNum('sellerFinanceDownPayment'); // New input


            const loanBalance = getNum('loanBalance');

            const loanPITI = getNum('loanPITI');


            // --- ANALYSIS ---

            // Expenses

            const vacancyCost = rent * vacancyPercent;

            const maintenanceCost = rent * maintenancePercent;

            const propManagementCost = rent * propManagementPercent;

            const totalOperatingExpenses = taxes + insurance + vacancyCost + maintenanceCost + propManagementCost;


            // Income

            const noi = rent - totalOperatingExpenses;

            

            // Max Allowable Payment for Creative Offer

            const maxAllowablePayment = noi - desiredCashflow;


            // --- OUTPUTS ---


            // 1. Cash Offer

            const maxCashOffer = (arv * 0.70) - repairs - closingCosts;

            document.getElementById('maxCashOffer').textContent = formatCurrency(maxCashOffer);


            // 2. Seller Finance Offer

            const sfOfferPrice = currentValue; 

            const sfLoanAmount = sfOfferPrice - sellerFinanceDownPayment; // Adjusted loan amount


            document.getElementById('sfOfferPrice').textContent = formatCurrency(sfOfferPrice);

            document.getElementById('sfDownPayment').textContent = formatCurrency(sellerFinanceDownPayment);

            document.getElementById('sfPayment').textContent = formatCurrency(maxAllowablePayment > 0 ? maxAllowablePayment : 0);

            document.getElementById('sfRate').textContent = `${sellerFinanceRate}%`;


            let sfTermYears = "N/A";

            if (maxAllowablePayment > 0 && sfLoanAmount > 0) {

                 if(sellerFinanceRate > 0) {

                    const monthlyRate = sellerFinanceRate / 100 / 12;

                    const pv = sfLoanAmount; // Use adjusted loan amount for calculation

                    const pmt = maxAllowablePayment; 


                    if (pmt > pv * monthlyRate) {

                        const numPayments = Math.log(pmt / (pmt - (pv * monthlyRate))) / Math.log(1 + monthlyRate);

                        sfTermYears = (numPayments / 12).toFixed(1);

                    } else {

                        sfTermYears = "∞"; 

                    }

                } else { // 0% interest case

                     sfTermYears = (sfLoanAmount / maxAllowablePayment / 12).toFixed(1);

                }

            } else if (sfLoanAmount <= 0) {

                 sfTermYears = "0.0"; // Paid off with down payment

            }

            document.getElementById('sfTerm').textContent = `${sfTermYears} yrs`;



            // 3. Subject-To Offer

            const subToOutputDiv = document.getElementById('subToOutput');

            const subToMessageDiv = document.getElementById('subToMessage');


            if (loanBalance > 0 && loanPITI > 0) {

                subToOutputDiv.classList.remove('hidden');

                subToMessageDiv.classList.add('hidden');

                

                const subToOperatingExpenses = vacancyCost + maintenanceCost + propManagementCost;

                const subToTotalExpenses = subToOperatingExpenses + loanPITI;

                const subToCashflow = rent - subToTotalExpenses;

                document.getElementById('subToCashflow').textContent = formatCurrency(subToCashflow);


            } else {

                subToOutputDiv.classList.add('hidden');

                subToMessageDiv.classList.remove('hidden');

            }

        }

        

        // Add event listeners to all inputs

        const inputs = document.querySelectorAll('input');

        inputs.forEach(input => input.addEventListener('input', calculate));


        // Initial calculation on page load

        window.onload = calculate;


    </script>
